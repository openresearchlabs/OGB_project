plot_taxon <- function(taxon) {
  
  # Check if the taxon exists in genus or species
  if (taxon %in% rowData(tse)$genus | taxon %in% rowData(tse)$species) {
    
    # Calculate signal (relative abundance)
        if (taxon %in% rowData(tse)$genus) {
          rownames(tse) <- rowData(tse)$genus
          df$signal <- as.vector(assay(tse, "relabundance")[taxon,])
        } else {
          rownames(tse) <- rowData(tse)$species
          df$signal <- as.vector(assay(tse, "relabundance")[taxon,])
        }
    
    daa_tab_tax <- daa_tab %>%
      filter(feature == taxon)
    
    anot_tax <- daa_tab_tax %>% 
        filter(value == "time")
    
    anot_tax$pval_color <- ifelse(anot_tax$qval_joint <= 0.25, "red", "blue")
    # Subset for before and after
    df_before <- df %>% filter(timepoint == "before")
    df_after <- df %>% filter(timepoint == "after")
    
    # Prepare data for plotting
        vars <- c("id", "timepoint", "signal")
        dfm <- melt(df[, vars])
        colnames(dfm) <- vars
        
        # Filter and mutate for paired design
        df <- df %>%
          filter(paired == "yes") %>%
          group_by(id) %>%
          mutate(
            change = if_else(timepoint == "after", signal - signal[timepoint == "before"], 0),
            change = if_else(timepoint == "before", change[timepoint == "after"], change)
          ) %>%
          ungroup() %>%
          as.data.frame(check.names = FALSE)
        
          min_change <- min(df$change, na.rm = TRUE)
          max_change <- max(df$change, na.rm = TRUE)
          median_change <- median(df$change, na.rm = TRUE)
    
    p12 <- ggplot(df, aes(x = diet, y = signal)) +
      geom_boxplot(outlier.shape = NA) +
      geom_point() + 
      scale_y_log10(labels = scales::percent) + 
      theme(text = element_text(size = 25), legend.position = "none") +
      labs(title = taxon, y = "Abundance (%)") +
      facet_wrap(~ timepoint, nrow = 1)
    
    p3 <- ggplot(df, aes(x = timepoint, y = signal)) +
      geom_boxplot(outlier.shape = NA) +
      geom_point() + 
      geom_line(aes(group = id, color = change)) +
      scale_color_gradient2(midpoint = median_change, low = "blue", mid = "white", high = "red", space = "Lab", limits = c(min_change, max_change)) +
      facet_wrap(~ diet) +
      scale_y_log10(labels = scales::percent) + 
      theme(text = element_text(size = 25), legend.position = "right") +
      labs(x = "Time", y = "Abundance (%)")
    
    # Combine the before and after plots
    return(wrap_plots(p12, p3, axes = "collect")
  )
  }
}
