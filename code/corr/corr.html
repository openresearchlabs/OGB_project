# Define variables and formulas
prevalence.th <- 10/100
detection.th <- 0.1/100
tax.level <- params$tax.level
tax.level <- "phylum_prevalent"

if (tax.level == "phylum_prevalent") {
    tse <- tse
} else {
    tse <- altExp(tse, tax.level)
}

Taxonomic level: phylum_prevalent

Taxa that are less prevalent than this (in relabundance) have been agglomerated:

  • Detection threshold: 0.1%
  • Prevalence threshold: 10%
  • Number of taxonomic groups: 10

microbiota vs. Short-chain fatty acids (12 SCFAs in total)

result1 <- getCrossAssociation(
    tse, assay.type1 = "relabundance_delta",
    col.var2 = c("scfa_sum"),
    test.signif = TRUE, show.warnings = FALSE, verbose = FALSE) 

result1 <- result1 %>%
    dplyr::rename(microbes = "Var1")

result <- result1 %>%
    select(-Var2, -pval)

datatable(
  result,
  options = list(pageLength = 10, dom = 'Bfrtip'),
  caption = "Correlation table of Short-chain fatty acids",
  rownames = FALSE
)  %>%

Heatmap

result <- rbind(result1, result2, result3, result4)

result <- result %>%
  mutate(
    signif_label = case_when(
      p_adj < 0.05 ~ "*",
      p_adj < 0.10 ~ "†",
      TRUE ~ ""
    ),
    label = ifelse(!is.na(cor), paste0(round(cor, 4), signif_label), "")
  )

result <- result[result$p_adj < 0.25, ]

# Plot
ggplot(result, aes(x = Var2, y = microbes, fill = cor)) +
  geom_tile(color = "white", size = 0.3) +
  geom_text(aes(label = label), size = 5) +
  scale_fill_gradient2(low = "blue", high = "red", mid = "white", midpoint = 0, na.value = "grey50") +
  theme_minimal(base_size = 14) +
  theme(text = element_text(family = "Roboto"), 
        axis.text = element_text(face = "bold"),
        axis.text.x = element_text(angle = 45, hjust = 1),
        panel.grid = element_blank()
  ) +
  labs(
    x = NULL, y = NULL, fill = "Correlation"
  ) + labs(caption = "p < 0.05 = *   p <0.10 = †")
