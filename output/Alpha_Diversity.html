Group-wise comparisons using Shannon index

# Define index and grouping variable
index <- "shannon"
group_var <- "group"

# Ensure the grouping variable is a factor
colData(tse)[[group_var]] <- factor(colData(tse)[[group_var]])

# Calculate p-values
pvals <- pairwise.wilcox.test(
  x = colData(tse)[[index]],
  g = colData(tse)[[group_var]],
  p.adjust.method = "fdr"
)

# Convert p-values to a data frame for ggpubr
pvals_df <- as.data.frame(pvals$p.value)
pvals_df$group1 <- rownames(pvals_df)
pvals_long <- pivot_longer(pvals_df, -group1, names_to = "group2", values_to = "p") %>%
  na.omit()

# Add y-axis position
pvals_long <- pvals_long %>%
  mutate(
    y.position = sapply(1:nrow(.), function(i) {
      max(c(
        colData(tse)[[index]][colData(tse)[[group_var]] == .$group1[i]],
        colData(tse)[[index]][colData(tse)[[group_var]] == .$group2[i]]
      ))
    })
  )
pvals_long$y.position <- max(pvals_long$y.position) + seq_len(nrow(pvals_long)) * 0.4
pvals_long$p <- round(pvals_long$p, 3)

# Filter for significant p-values (p < 0.05)
pvals_long <- pvals_long %>% filter(p < 0.05)

# Visualization using ggplot2 - boxplot with jitter plot (no color)
library(ggplot2)
df <- as.data.frame(colData(tse))  # Convert colData to a data frame for ggplot2

p <- ggplot(df, aes(x = .data[[group_var]], y = .data[[index]])) +
  geom_boxplot(outlier.shape = NA, alpha = 0.7) +  # Add boxplot
  geom_jitter(width = 0.2, size = 1.5, alpha = 0.8) +  # Add jittered points
  stat_pvalue_manual(pvals_long, label = "p") +  # Add significant p-value annotations
  labs(
    x = "Groups",
    y = "Shannon Index"
  ) +
  theme_minimal() +
  theme(
    axis.text.x = element_text(angle = 45, hjust = 1),  # Rotate X-axis labels
    text = element_text(size = 12)
  )

print(p)
