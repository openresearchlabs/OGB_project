Author

Nitin Bayal

Differential Abundance Analysis

This document presents the Differential Abundance Analysis (DAA) using the ANCOMBC-II method. The analysis identifies significant features between diet groups and visualizes the top findings.

Data

# Load functions
source("funct.R")

# Load TreeSummarizedExperiment object
tse <- readRDS("../data/tse.Rds")

# Preprocess data
# Agglomerate at genus level and filter by prevalence
tse <- agglomerateByPrevalence(
  tse, rank = "genus",
  prevalence = 0.1,
  assay.type = "counts"
)

# Transform count assay to relative abundances
tse <- transformAssay(
  tse, assay.type = "counts",
  method = "relabundance"
)

# Define the variable of interest
group_var <- "diet"

ANCOM-BC II

# Perform ANCOMBC-II
daa_results <- ancombc2(
  data = assay(tse, "relabundance"),
  meta_data = colData(tse),
  group = group_var,
  fix_formula = "diet",
  p_adj_method = "fdr",
  prv_cut = 0.05,
  lib_cut = 0,
  struc_zero = TRUE,
  neg_lb = TRUE,
  alpha = 0.05,
  global = TRUE
)

# Extract significant features
significant_results <- daa_results$res %>%
  filter(q_diet2 < 0.05) %>%  # Filter based on adjusted p-values
  arrange(q_diet2)  # Sort by adjusted p-value

Results table
